# How to analyze
# Run on samples
dotnet run --project src/StackProcessing.Probe/StackProcessing.Probe.fsproj -- samples --json --repeat 3 -j 6
# measure analytics
dotnet run --project src/StackProcessing.Probe/StackProcessing.Probe.fsproj -- analysis
# generate boiler plate additions
dotnet run --project src/StackProcessing.Probe/StackProcessing.Probe.fsproj -- probing tmp/analysis/probing-boilerplate.json --analysis-features tmp/analysis/featureDiagnostics.csv --low-support-threshold 2 --emit-json tmp/probingGraphs
# run on new
dotnet run --project src/StackProcessing.Probe/StackProcessing.Probe.fsproj -- samples --json --repeat 3 --extra-json-only --extra-json-root tmp/probingGraphs -j 6
# update analysis
dotnet run --project src/StackProcessing.Probe/StackProcessing.Probe.fsproj -- analysis --extra-json-root tmp/probingGraphs
